```ts
import { z } from "zod";
import { sqliteTable, text, integer, primaryKey } from "drizzle-orm/sqlite-core";

    runId: text("run_id").notNull(),

  description: z.string(),
```

```sql
  run_id TEXT NOT NULL,

CREATE TABLE IF NOT EXISTS implement (
  run_id TEXT NOT NULL,
  node_id TEXT NOT NULL,
  iteration INTEGER NOT NULL DEFAULT 0,
  summary TEXT NOT NULL,
  files_changed TEXT,
  PRIMARY KEY (run_id, node_id, iteration)
);

CREATE TABLE IF NOT EXISTS report (
  run_id TEXT NOT NULL,
  node_id TEXT NOT NULL,
  summary TEXT NOT NULL,
  PRIMARY KEY (run_id, node_id)
```

```ts
// db.ts
import { resolve } from "node:path";
import { Database } from "bun:sqlite";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { migrate } from "drizzle-orm/bun-sqlite/migrator";
import { sqliteTable, text } from "drizzle-orm/sqlite-core";

import { discoverTable } from "./components/Discover.schema";
import { implementTable } from "./components/Implement.schema";
import { reportTable } from "./components/Report.schema";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
});

export const schema = {
  input: inputTable,
  discover: discoverTable,
  implement: implementTable,
  output: reportTable,
};

const DB_PATH = resolve(new URL("./workflow.db", import.meta.url).pathname);
const MIGRATIONS_PATH = resolve(new URL("./drizzle", import.meta.url).pathname);

const sqlite = new Database(DB_PATH);
sqlite.exec("PRAGMA journal_mode = WAL");
sqlite.exec("PRAGMA foreign_keys = ON");

export const db = drizzle(sqlite, { schema });

migrate(db, { migrationsFolder: MIGRATIONS_PATH });
```

**4) Helpers**

```ts
// lib/render.ts
import React from "react";
import { renderToStaticMarkup } from "react-dom/server";
import type { MDXContent } from "mdx/types";

export function render(Component: MDXContent, props: Record<string, any> = {}): string {
  const html = renderToStaticMarkup(React.createElement(Component, props));
  return html
    .replace(/<\/?(p|div|h[1-6]|li|ul|ol|pre|blockquote|section)[^>]*>/gi, "\n")
    .replace(/<br\s*\/?>(?!\n)/gi, "\n")
    .replace(/<[^>]+>/g, "")
    .replace(/&amp;/g, "&")
    .replace(/&lt;/g, "<")
    .replace(/&gt;/g, ">")
    .replace(/&quot;/g, '"')
    .replace(/&#x27;/g, "'")
    .replace(/&#x2F;/g, "/")
    .replace(/\n{3,}/g, "\n\n")
    .trim();
}
```

```ts
// lib/ctx.ts
import type { SmithersCtx } from "smithers";
import type { SQLiteTable } from "drizzle-orm/sqlite-core";

export type WorkflowCtx = SmithersCtx<any>;

export function typedOutput<T>(ctx: WorkflowCtx, table: SQLiteTable, args: { nodeId: string }): T | undefined {
  const rows = (ctx.outputs as any)(table) as any[] | undefined;
  if (!rows?.length) return undefined;
  let latest: any;
  let latestIteration = -Infinity;
  for (const row of rows) {
    if (!row || row.nodeId !== args.nodeId) continue;
    const iter = Number.isFinite(Number(row.iteration)) ? Number(row.iteration) : 0;
    if (!latest || iter >= latestIteration) {
      latest = row;
      latestIteration = iter;
    }
  }
  return latest as T | undefined;
}
```

```ts
// lib/coerce.ts
import { z } from "zod";

export function coerceJsonArray<T>(raw: unknown, item: z.ZodType<T>): T[] {
  if (raw == null) return [];
  let value = raw;
  if (typeof value === "string") {
    try { value = JSON.parse(value); } catch { return []; }
  }
  const result = z.array(item).safeParse(value);
  return result.success ? result.data : [];
}
```

```tsx
// components/Discover.tsx
import { Task } from "smithers";
import { render } from "../lib/render";
import { claude } from "../agents";
import DiscoverPrompt from "../prompts/discover.mdx";
import { discoverTable, discoverOutputSchema } from "./Discover.schema";

const schemaExample = {
  tickets: [{ id: "ticket-id", title: "Title", description: "Details" }],
};

export function Discover() {
  const prompt = render(DiscoverPrompt, {
    schema: JSON.stringify(schemaExample, null, 2),
  });

  return (
    <Task id="discover" output={discoverTable} outputSchema={discoverOutputSchema} agent={claude}>
      {prompt}
    </Task>
  );
}
```

```tsx
// components/Implement.tsx
import { Task } from "smithers";
import { render } from "../lib/render";
import { claude } from "../agents";
import ImplementPrompt from "../prompts/implement.mdx";
import { implementTable, implementOutputSchema } from "./Implement.schema";

export function Implement({ ticket }: { ticket: { id: string; title: string; description: string } }) {
  const prompt = render(ImplementPrompt, { ticket });

  return (
    <Task
      id={`${ticket.id}:implement`}
      output={implementTable}
      outputSchema={implementOutputSchema}
      agent={claude}
    >
      {prompt}
    </Task>
  );
}
```

**6) Workflow**

```tsx
// workflow.tsx
import { smithers, Workflow, Sequence } from "smithers";
import { db } from "./db";
import { Discover } from "./components/Discover";
import { Implement } from "./components/Implement";
import { Report } from "./components/Report";
import { discoverTable, ticketSchema } from "./components/Discover.schema";
import { coerceJsonArray } from "./lib/coerce";
import { typedOutput } from "./lib/ctx";

export default smithers(db, (ctx) => {
  const discover = typedOutput<{ tickets: unknown }>(ctx, discoverTable, { nodeId: "discover" });
  const tickets = coerceJsonArray(discover?.tickets, ticketSchema);
  const needsDiscovery = tickets.length === 0;

  return (
    <Workflow name="tutorial-workflow">
      <Sequence>
        {needsDiscovery ? <Discover /> : null}
        {tickets.map((ticket) => (
          <Implement key={ticket.id} ticket={ticket} />
        ))}
        <Report summary={`Implemented ${tickets.length} ticket(s).`} />
      </Sequence>
    </Workflow>
  );
});
```

**7) Run the workflow**

```bash
bun install
bun run smithers run workflow.tsx --input '{}'
```

Notes:

- `skipIf` is evaluated at render time. Use conditional rendering or sequencing instead of `skipIf` for tasks that should run after prerequisites.
- Smithers validates outputs against your Drizzle table schema. Zod schemas are still useful for prompt templates and JSON examples.
