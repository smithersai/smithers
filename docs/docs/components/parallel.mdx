---
title: <Parallel>
description: Execute children concurrently.
---

```tsx
// workflow.tsx
import { smithers, Workflow, Task, Parallel } from "smithers";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, integer, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
});

const lintTable = sqliteTable(
  "lint",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    errors: integer("errors").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const testTable = sqliteTable(
  "test",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    passed: integer("passed", { mode: "boolean" }).notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = { input: inputTable, output: testTable, lint: lintTable, test: testTable };
const db = drizzle("./workflow.db", { schema });

export default smithers(db, () => (
  <Workflow name="parallel">
    <Parallel maxConcurrency={4}>
      <Task id="lint" output={schema.lint}>
        {{ errors: 0 }}
      </Task>
      <Task id="test" output={schema.test}>
        {{ passed: true }}
      </Task>
    </Parallel>
  </Workflow>
));
```

**Props**

| Prop | Type | Description |
| --- | --- | --- |
| `maxConcurrency` | `number` | Max parallel tasks (default ∞) |
| `skipIf` | `boolean` | Skip all children if true |
| `children` | `ReactNode` | Child nodes |
