---
title: <Task>
description: A single executable node.
---

`<Task>` represents one unit of work.

```tsx
// workflow.tsx
import { smithers, Workflow, Task } from "smithers";
import { agent } from "ai";
import { anthropic } from "@ai-sdk/anthropic";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
  description: text("description").notNull(),
});

const analyzeTable = sqliteTable(
  "analyze",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    summary: text("summary").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const setupTable = sqliteTable(
  "setup",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    files: text("files", { mode: "json" }).$type<string[]>(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = {
  input: inputTable,
  output: analyzeTable,
  analyze: analyzeTable,
  setup: setupTable,
};

const db = drizzle("./workflow.db", { schema });

const codeAgent = agent({
  model: anthropic("claude-sonnet-4-20250514"),
  system: "You are a senior software engineer.",
});

export default smithers(db, (ctx) => (
  <Workflow name="tasks">
    <Task id="analyze" output={schema.analyze} agent={codeAgent}>
      {`Analyze: ${ctx.input.description}`}
    </Task>
    <Task id="setup" output={schema.setup}>
      {{ files: ["README.md", "package.json"] }}
    </Task>
  </Workflow>
));
```

**Props**

| Prop | Type | Description |
| --- | --- | --- |
| `id` | `string` | Stable node identity (required) |
| `output` | `Table` | Drizzle table for this node's output |
| `outputSchema` | `ZodObject` | Optional Zod schema for agent output |
| `agent` | `Agent` | AI SDK agent (if provided, children → prompt) |
| `skipIf` | `boolean` | Skip execution if true |
| `needsApproval` | `boolean` | Require approval before running |
| `timeoutMs` | `number` | Max duration in ms before failing |
| `retries` | `number` | Retry count on failure (default 0) |
| `continueOnFail` | `boolean` | Continue workflow if task fails |
| `label` | `string` | Optional label for UI/metadata |
| `meta` | `Record<string, unknown>` | Optional metadata |
| `children` | `string | output` | Prompt (agent) or output object (static) |
