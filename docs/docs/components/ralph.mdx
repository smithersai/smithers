---
title: <Ralph>
description: Iterative loops with a stop condition.
---

```tsx
// workflow.tsx
import { smithers, Workflow, Task, Ralph } from "smithers";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, integer, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
});

const reviewTable = sqliteTable(
  "review",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    iteration: integer("iteration").notNull().default(0),
    approved: integer("approved", { mode: "boolean" }).notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId, t.iteration] }),
  }),
);

const schema = { input: inputTable, output: reviewTable, review: reviewTable };
const db = drizzle("./workflow.db", { schema });

export default smithers(db, (ctx) => (
  <Workflow name="ralph">
    <Ralph
      until={ctx.output(schema.review, { nodeId: "review" }).approved}
      maxIterations={3}
      onMaxReached="return-last"
    >
      <Task id="review" output={schema.review}>
        {{ approved: true }}
      </Task>
    </Ralph>
  </Workflow>
));
```

**Props**

| Prop | Type | Description |
| --- | --- | --- |
| `id` | `string` | Optional loop id (useful for multiple loops) |
| `until` | `boolean` | Stop when true |
| `maxIterations` | `number` | Max iterations (default 5) |
| `onMaxReached` | `"fail" | "return-last"` | Behavior at max (default return-last) |
| `skipIf` | `boolean` | Skip loop if true |
| `children` | `ReactNode` | Child nodes |
