---
title: Context API
description: Read inputs and outputs in a type-safe way.
---

Smithers passes a `ctx` object into your workflow builder.

```tsx
// workflow.tsx
import { smithers, Workflow, Task } from "smithers";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
  description: text("description").notNull(),
});

const analyzeTable = sqliteTable(
  "analyze",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    summary: text("summary").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = { input: inputTable, output: analyzeTable, analyze: analyzeTable };
const db = drizzle("./workflow.db", { schema });

export default smithers(db, (ctx) => {
  const analyzeRow = ctx.outputMaybe(schema.analyze, { nodeId: "analyze" });

  return (
    <Workflow name="ctx">
      <Task id="analyze" output={schema.analyze}>
        {{ summary: `Analyze: ${ctx.input.description}` }}
      </Task>

      {analyzeRow ? (
        <Task id="summary" output={schema.output}>
          {{ summary: analyzeRow.summary }}
        </Task>
      ) : null}
    </Workflow>
  );
});
```

`ctx` fields:

- `runId`: current run ID
- `iteration`: current loop iteration (inside `<Ralph>`)
- `iterations`: per-Ralph iteration map (keyed by Ralph id)
- `input`: input row (typed from `schema.input`)
- `outputs`: table accessor (array per table)

Access outputs:

- `ctx.output(table, { nodeId, iteration? })` throws if missing
- `ctx.outputMaybe(table, { nodeId, iteration? })` returns `undefined` if missing
- `ctx.outputs.<table>` returns all rows for that table
