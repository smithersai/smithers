---
title: Dynamic Plan
description: Branch based on an analysis result.
---

```tsx
// workflow.tsx
import { smithers, Workflow, Task, Sequence, Branch } from "smithers";
import { agent } from "ai";
import { anthropic } from "@ai-sdk/anthropic";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
  description: text("description").notNull(),
});

const analysisTable = sqliteTable(
  "analysis",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    summary: text("summary").notNull(),
    complexity: text("complexity").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const planTable = sqliteTable(
  "plan",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    steps: text("steps", { mode: "json" }).$type<string[]>(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const implementTable = sqliteTable(
  "implement",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    result: text("result").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const outputTable = sqliteTable(
  "output",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    summary: text("summary").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = {
  input: inputTable,
  output: outputTable,
  analysis: analysisTable,
  plan: planTable,
  implement: implementTable,
};

const db = drizzle("./workflow.db", { schema });

const analyzer = agent({
  model: anthropic("claude-sonnet-4-20250514"),
  system: "Return JSON with { summary: string, complexity: 'low' | 'high' }",
});

const planner = agent({
  model: anthropic("claude-sonnet-4-20250514"),
  system: "Return JSON with { steps: string[] }",
});

const implementer = agent({
  model: anthropic("claude-sonnet-4-20250514"),
  system: "Return JSON with { result: string }",
});

export default smithers(db, (ctx) => {
  const analysis = ctx.outputMaybe(schema.analysis, { nodeId: "analyze" });
  const isHigh = analysis?.complexity === "high";

  return (
    <Workflow name="dynamic-plan">
      <Sequence>
        <Task id="analyze" output={schema.analysis} agent={analyzer}>
          {`Analyze: ${ctx.input.description}. Return JSON with summary and complexity.`}
        </Task>

        {analysis ? (
          <Branch
            if={isHigh}
            then={
              <Sequence>
                <Task id="plan" output={schema.plan} agent={planner}>
                  {`Create a plan for: ${analysis.summary}`}
                </Task>
                <Task id="implement" output={schema.implement} agent={implementer}>
                  {`Implement using steps: ${ctx.outputMaybe(schema.plan, { nodeId: "plan" })?.steps ?? []}`}
                </Task>
              </Sequence>
            }
            else={
              <Task id="implement" output={schema.implement} agent={implementer}>
                {`Quick fix: ${analysis.summary}`}
              </Task>
            }
          />
        ) : null}

        <Task id="summary" output={schema.output}>
          {{ summary: ctx.outputMaybe(schema.implement, { nodeId: "implement" })?.result ?? "" }}
        </Task>
      </Sequence>
    </Workflow>
  );
});
```

Run it:

```bash
smithers run workflow.tsx --input '{"description":"Fix auth bugs"}'
```
