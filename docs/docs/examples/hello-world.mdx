```tsx
// workflow.tsx
import { smithers, Workflow, Task, Sequence } from "smithers";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
  name: text("name").notNull(),
});

const greetingTable = sqliteTable(
  "greeting",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    message: text("message").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const outputTable = sqliteTable(
  "output",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    message: text("message").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = {
  input: inputTable,
  output: outputTable,
  greeting: greetingTable,
};

const db = drizzle("./workflow.db", { schema });

export default smithers(db, (ctx) => {
  const greeting = ctx.outputMaybe(schema.greeting, { nodeId: "greet" });

  return (
    <Workflow name="hello">
      <Sequence>
        <Task id="greet" output={schema.greeting}>
          {{ message: `Hello, ${ctx.input.name}!` }}
        </Task>
        <Task id="summary" output={schema.output}>
          {{ message: greeting?.message ?? "Hello" }}
        </Task>
      </Sequence>
    </Workflow>
  );
});
```

Run it:

```bash
smithers run workflow.tsx --input '{"name":"World"}'
```
