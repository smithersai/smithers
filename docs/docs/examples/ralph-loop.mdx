---
title: Ralph Loop
description: Iterate until a condition is met.
---

```tsx
// workflow.tsx
import { smithers, Workflow, Task, Sequence, Ralph } from "smithers";
import { agent } from "ai";
import { anthropic } from "@ai-sdk/anthropic";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, integer, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
  description: text("description").notNull(),
});

const reviewTable = sqliteTable(
  "review",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    iteration: integer("iteration").notNull().default(0),
    approved: integer("approved", { mode: "boolean" }).notNull(),
    notes: text("notes").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId, t.iteration] }),
  }),
);

const outputTable = sqliteTable(
  "output",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    finalNotes: text("final_notes").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = { input: inputTable, output: outputTable, review: reviewTable };
const db = drizzle("./workflow.db", { schema });

const reviewer = agent({
  model: anthropic("claude-sonnet-4-20250514"),
  system: "Return JSON with { approved: boolean, notes: string }",
});

export default smithers(db, (ctx) => (
  <Workflow name="ralph-loop">
    <Ralph
      until={ctx.outputMaybe(schema.review, { nodeId: "review" })?.approved ?? false}
      maxIterations={3}
      onMaxReached="return-last"
    >
      <Sequence>
        <Task id="review" output={schema.review} agent={reviewer}>
          {`Review: ${ctx.input.description}. Return JSON with approved and notes.`}
        </Task>
      </Sequence>
    </Ralph>

    <Task id="summary" output={schema.output}>
      {{ finalNotes: ctx.outputMaybe(schema.review, { nodeId: "review" })?.notes ?? "" }}
    </Task>
  </Workflow>
));
```

Run it:

```bash
smithers run workflow.tsx --input '{"description":"Fix auth bugs"}'
```
