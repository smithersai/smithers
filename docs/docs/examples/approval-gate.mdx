---
title: Approval Gate
description: Require human approval before continuing.
---

```tsx
// workflow.tsx
import { smithers, Workflow, Task, Sequence } from "smithers";
import { drizzle } from "drizzle-orm/bun-sqlite";
import { sqliteTable, text, integer, primaryKey } from "drizzle-orm/sqlite-core";

const inputTable = sqliteTable("input", {
  runId: text("run_id").primaryKey(),
  title: text("title").notNull(),
});

const draftTable = sqliteTable(
  "draft",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    content: text("content").notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const publishTable = sqliteTable(
  "publish",
  {
    runId: text("run_id").notNull(),
    nodeId: text("node_id").notNull(),
    url: text("url").notNull(),
    approved: integer("approved", { mode: "boolean" }).notNull(),
  },
  (t) => ({
    pk: primaryKey({ columns: [t.runId, t.nodeId] }),
  }),
);

const schema = { input: inputTable, output: publishTable, draft: draftTable, publish: publishTable };
const db = drizzle("./workflow.db", { schema });

export default smithers(db, (ctx) => (
  <Workflow name="approval">
    <Sequence>
      <Task id="draft" output={schema.draft}>
        {{ content: `Draft: ${ctx.input.title}` }}
      </Task>
      <Task id="publish" output={schema.publish} needsApproval>
        {{ url: "https://example.com/post", approved: true }}
      </Task>
    </Sequence>
  </Workflow>
));
```

Run it:

```bash
smithers run workflow.tsx --input '{"title":"Launch Plan"}'
```

When the run pauses for approval:

```bash
smithers approve workflow.tsx --run-id <id> --node-id publish
# or
smithers deny workflow.tsx --run-id <id> --node-id publish
```
